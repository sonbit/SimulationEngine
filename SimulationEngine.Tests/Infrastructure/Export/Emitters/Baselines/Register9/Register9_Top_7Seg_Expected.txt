	output [3:0] an

	wire [1:0] Q;

	c_Register9 c_Register9 (
		.RdAddr1(sw[1:0]),

		.WrAddr0(sw[7:6]),
		.Clk(sw[8]),
		.WrData(sw[10:9]),

	);

	basys3_7segment_display #(
		.CLK_HZ(100_000_000),
		.SCAN_HZ(1000)
	) basys3_7segment_display (
		.clk(clk),
		.rst_n(1'b1),
		.digit3_mst(2'b01),
		.digit3_lst(Q),
		.digit2_mst(2'b01),
		.digit2_lst(2'b01),
		.digit1_mst(2'b01),
		.digit1_lst(2'b01),
		.digit0_mst(2'b01),
		.digit0_lst(2'b01),
		.enable_mask(4'b1000),
		.seg(seg),
		.dp(dp),
		.an(an)
	);

	assign led[1:0] = Q;
endmodule
