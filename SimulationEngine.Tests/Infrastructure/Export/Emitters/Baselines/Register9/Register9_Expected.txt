module f_ZD0PPPPPP_latch (
	input wire [1:0] C,
	input wire [1:0] B,
	input wire [1:0] A,
	output reg [1:0] Q
);
	always @(posedge C[1]) begin
		Q <=
			(A == 2'b01) ? 2'b01 :
			(A == 2'b10) ? 2'b10 :
			2'b11;
	end
endmodule
